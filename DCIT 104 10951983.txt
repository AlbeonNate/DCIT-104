#include <bits/stdc++.h>
using namespace std;

// Function to calculate the average
// of even numbers
int averageEven(int h)
{
	if (h % 2 != 0) {
		cout<<"Invalid Input";
		return -1;
	}

	return (h + 2) / 2;
}

// driver function
int main()
{
	int h = 16;
	cout<<averageEven(h)<<endl;
	return 0;
}
